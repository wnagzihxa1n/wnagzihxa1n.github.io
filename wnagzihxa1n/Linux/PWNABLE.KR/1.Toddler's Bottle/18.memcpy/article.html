这里是在对比slow_memcpy和fast_memcpy两个函数的执行速度

第一个是slow_memcpy函数，最慢的循环拷贝

char *slow_memcpy(char *dest, const char *src, size_t len)
{
    int i;
    for (i = 0; i < len; i++)
    {
        dest[i] = src[i];
    }
    return dest;
}

第二个是fast_memcpy函数，使用的是内联汇编，

char *fast_memcpy(char *dest, const char *src, size_t len)
{
    size_t i;
    // 64-byte block fast copy
    if (len >= 64)
    {
        i = len / 64;
        len &= (64 - 1);
        while (i-- > 0)
        {
            __asm__ __volatile__(
                "movdqa (%0), %%xmm0\n"
                "movdqa 16(%0), %%xmm1\n"
                "movdqa 32(%0), %%xmm2\n"
                "movdqa 48(%0), %%xmm3\n"
                "movntps %%xmm0, (%1)\n"
                "movntps %%xmm1, 16(%1)\n"
                "movntps %%xmm2, 32(%1)\n"
                "movntps %%xmm3, 48(%1)\n" ::"r"(src),
                "r"(dest)
                : "memory");
            dest += 64;
            src += 64;
        }
    }

    // byte-to-byte slow copy
    if (len)
        slow_memcpy(dest, src, len);
    return dest;
}

那么我们来看一下这条指令为什么会出错

=> 0x80487cc <fast_memcpy+52>:    movntps XMMWORD PTR [edx],xmm0
EDX: 0x804c4a8 --> 0x0

搜索这条指令，找到这条指令的说明

movntps m128,XMM
m128 <== XMM 直接把XMM中的值送入m128，不经过cache,必须对齐16字节.

我们可以看到EDX寄存器的值并没有16字节对齐，所以报异常了

那也就是说，我们每次申请的空间大小要16字节对齐

所以，一个简单的脚本就可以了

#!/usr/bin/env python
# coding = utf-8

import socket
import re
import math

buffer_size = 2048

def main():
    target_addr = ('pwnable.kr', 9022)
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.connect(target_addr)
    sock.recv(buffer_size)
    sock.recv(buffer_size)
    for i in range(4, 14):
        range_left = int(math.pow(2, i - 1))
        range_right = int(math.pow(2, i))
        for j in range(range_left, range_right + 1):
            if ((j + 4) % 16 > 8) or ((j + 4) % 16 == 0):
                sock.send(str(j) + "\n")
                break
        data = sock.recv(buffer_size)
        print data
    sock.recv(buffer_size)
    flag = sock.recv(buffer_size + 100)
    print flag
    sock.close()

if __name__ == '__main__':
    main()
